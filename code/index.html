               var index = 0;
               var arrayOfImages = [
                   "http://logok.org/wp-content/uploads/2015/04/Microsoft-Edge-Browser-logo.png",

               function MoveLeft() {
                   if (index == 0) {
                       index = 3
                   }
                   index--;
                   var img = document.getElementById("imgRotate")
                   img.src = arrayOfImages[index]
       
               }
       
               function MoveRight() {
                   if (index == 2) {
                       index = -1
                   }
                   index++
                   var img = document.getElementById("imgRotate")
                   img.src = arrayOfImages[index]
               }
